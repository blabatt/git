\section{Patching}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection*{Partial changes}
Use -{}-patch with: git [add|reset|checkout|stash]: \\
\code{git add -{}-patch} \# interactive patch staging

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection*{git format-patch}
Generate emailable patches, one per commit 
past indicated branch (here origin/master): \\
\code{git format-patch -M origin/master}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection*{git [send]}
\code{cat *.patch | git imap-send }\\
\code{git send-email *.patch}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection*{git [apply|am]}
{\footnotesize Use former if user submitted diff format \say{patch}; latter if user submitted format-patch patches.  Latter is better: already contains commit metadata.}\\
\code{git apply /tmp/pname.patch}\\
\code{git am pname.patch}\\
\code{git am -3 pname.patch} \# attempt 3-way merge
